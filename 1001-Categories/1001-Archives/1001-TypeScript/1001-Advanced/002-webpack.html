# webpack配置（webpack.config.js）

当使用webpack打包的时候，如果没有跟--config参数，则默认添加--config webpack.config.js

# 全局安装
webpack -v
# 本地安装
npx webpack -v

# 根据`webpack.config.dev.js`打包
npx webpack --config webpack.config.dev.js
# 根据`webpack.config.js`打包
npx webpack
npx webpack --config webpack.config.js

# 入口（entry）

// 单独入口
modeul.exports = {
    entry: './src/index.ts'
}

// 多入口（代码分离）
modeul.exports = {
    entry: {
        index: './src/index.ts',
        index2: './src/index2.ts'
    }
}

// 多入口（代码分离 + 入口依赖-多入口引入相同依赖项）
modeul.exports = {
    entry: {
        index: {
            import: './src/index.ts',
            dependOn: 'shared'
        },
        index2: {
            import: './src/index2.ts',
            dependOn: 'shared'
        },
        shared: './src/alib.ts',
    }
}

loader可以通过include和exclude来缩小转换时的检测

比如ts-loader

module.exports = {
    module: {
        rules: [
            {
                test: /\.tsx?$/,
                loader: 'ts-loader',
                // 缩小检测范围
                include: path.resolve(__dirname, 'src'),
                options: {
                    // 开启后会关闭类型检测，缩短`ts-loader`构建时间
                    // 如果需要类型检测，可以使用`ForkTsCheckerWebpackPlugin`把类型检查过程移到单独的进程
                    transpileOnly: true
                }
            }
        ]
    } 
    }

# npm scripts

// package.json
{
  "private": true,
  "scripts": {
      // 编译
    "build": "webpack",
    // 自动编译

    // 自动构建+启动本地服务器+live reloading(实时重新加载)
    // 需要安装 `webpack-dev-server`本地依赖

  }
}

# 开发环境

  • webpack --watch：自动构建, 需要刷新浏览器才能看到最新结果
  • webpack-dev-server： 自动构建+启动本地服务器+live reloading（实时重新加载），无需刷新浏览器
  • express + webpack-dev-middleware：自动构建+启动本地服务器

# webpack-dev-server

const path = require('path');

module.exports = {
    output: {
        filename: '[name].[contenthash].js',
        path: path.resolve(__dirname, 'dist'),
        clean: true,
        publicPath: '/'
    },
    resolve: {
        extensions: ['.tsx', '.ts', '.js']
    },
    devtool: 'inline-source-map',
    devServer: {
        // 将 dist 目录下的文件 serve 到 localhost:8080
        contentBase: './dist',

    }
}

# express + webpack-dev-middleware

需要安装依赖: npm install --save-dev express webpack-dev-middleware

新建server.js

const express = require('express');
const webpack = require('webpack');
const webpackDevMiddleware = require('webpack-dev-middleware');

const app = express();
const config = require('./webpack.config.js');
const compiler = webpack(config);

// 告知 express 使用 webpack-dev-middleware，
// 以及将 webpack.config.js 配置文件作为基础配置。
app.use(
  webpackDevMiddleware(compiler, {
    publicPath: config.output.publicPath,
  })
);

// 将文件 serve 到 port 3000。
app.listen(3000, function () {
  console.log('Example app listening on port 3000!\n');
});

# 更多

常用的loader：ts-loader, style-loader, css-loader

常用plugin：

  • HtmlWebpackPlugin，创建html
  • ForkTsCheckerWebpackPlugin ts类型检测，把类型检查过程移到单独的进程
  • WebpackManifestPlugin mainfest

构建方式：

  • npx webpack: 构建
  • npx webpack --watch: 构建并自动编译

最基本的配置：

const HtmlWebpackPlugin = require('html-webpack-plugin');
const path = require('path');
const pkg = require('./package.json');

module.exports = (env) => {
    const publicPath = env.WEBPACK_SERVE === true ? '/' : '.'
    const mode = process.env.NODE_ENV || 'development';
    return {
        mode,
        entry: {
            app: path.resolve(__dirname, './src/app.ts')
        },
        output: {
            filename: '[name].[contenthash].js',
            path: path.resolve(__dirname, 'dist'),
            clean: true,
            publicPath
        },
        resolve: {
            extensions: ['.tsx', '.ts', '.js']
        },
        devtool: 'inline-source-map',
        devServer: {
            contentBase: './dist',
            hot: true
        },
        plugins: [
            new HtmlWebpackPlugin({
                title: pkg.name,
                // 如果有html模版可以启用这里
                // template: './src/index.html'
            })
        ],
        module: {
            rules: [
                {
                    test: /\.tsx?$/,
                    loader: 'ts-loader',
                    include: path.resolve(__dirname, 'src'),
                    options: {
                        transpileOnly: true
                    }
                }
                ,
                {
                    test: /\.css$/i,
                    use: ['style-loader', 'css-loader']
                }
                ,
                {
                    test: /\.(json)$/i,
                    type: 'asset/resource'
                }
                ,
                {
                    test: /\.(woff|woff2|eot|ttf|otf)$/i,
                    type: 'asset/resource'
                }
                ,
                {
                    test: /\.(png|svg|jpg|jpeg|git)$/i,
                    type: 'asset/resource'
                }
            ]
        }
    };
};
